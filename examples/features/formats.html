Currency Format

Parameter Default Description
thousandSeparator '.' Thousand separator
decimalSeparator ',' Decimal separator
decimalPlaces 0 Number of decimal places
suffix '' Suffix 
// Format is inline in column config:
columns: {
    0: { 
        name: 'salary', 
        type: 'currency', 
        format: {
            thousandSeparator: '.',
            prefix: '$ '
        }
    }
}

Input: 120000

$ 50.000.000

Percentage Format

// Format is inline in column config:
columns: {
    0: { 
        name: 'score', 
        type: 'percentage', 
        format: { suffix: '%' }
    }
}

Input: 85

85%

Date Format

Token Example Description
DD 01, 15, 31 2-digit Day
MM 01, 06, 12 2-digit Month
YYYY 2024, 2025 4-digit Year 
// Format is inline in column config:
columns: {
    // Custom Format
    0: { name: 'birthdate', type: 'date', format: { format: 'DD/MM/YYYY' } },
    
    // ISO Format
    1: { name: 'joined', type: 'date', format: { format: 'YYYY-MM-DD' } }
}
